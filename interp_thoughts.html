PyKEP ephemeris interpolation initial thoughts¶

Chris Andre (github.com/ChrisAndre, chrisandre01@gmail.com)¶

The goal is to find an interpolating function over the ephemerides taken from a high-fidelity SPICE planet object which can then be distributed so that the high-quality ephemerides can be reconstructed across many other nodes without needing the heavy kernels to be distributed.

This is a preliminary Python study of the interpolation part... I am making sure there will not be any issues with the interpolation.

class Polynomial(object):
    """
    Polynomial defined by c[i] x^i + ...
    """
    def __init__(self, coeff):
        self.coeff = coeff
    def __str__(self):
        string = 'y(x) = '
        for i in range(len(self.coeff) - 1, 0, -1):
            string += '({:.2e})'.format(self.coeff[i]) + '*x^%i + ' % i
        string += '{:.2e}'.format(self.coeff[0])
        return string
    def scaled(self, a):
        ret = []
        for i in range(len(self.coeff)):
            ret.append(self.coeff[i] * a)
        return Polynomial(ret)
    def multiply(self, poly):
        new_coeff_sz = len(self.coeff) + len(poly.coeff) - 1
        new_coeff = [0] * new_coeff_sz
        for i in range(len(self.coeff)):
            for j in range(len(poly.coeff)):
                power = i+j
                new_coeff[power] += self.coeff[i] * poly.coeff[j]
        return Polynomial(new_coeff)
    def add(self, poly):
        if len(self.coeff) > len(poly.coeff):
            large = self
            small = poly
        else:
            large = poly
            small = self
        new_coeff = [0] * len(large.coeff)
        for i in range( len(small.coeff) ):
            new_coeff[i] = self.coeff[i] + poly.coeff[i]
        for i in range( len(small.coeff), len(large.coeff) ):
            new_coeff[i] = large.coeff[i]
        return Polynomial(new_coeff)
    def evl(self, x):
        ret = 0.0
        x_ = 1.0
        for i in range(len(self.coeff)):
            ret += self.coeff[i] * x_
            x_ *= x
        return ret

class Interpolator(object):
    """
    Lagrangian interpolation method
    """
    def __init__(self, x_vals, y_vals):
        # lengths should be equal
        self.x_vals = x_vals
        self.y_vals = y_vals
        self.setPolynomial()
    def getY(self, x):
        return self.poly.evl(x)
    def setPolynomial(self):
        ret = Polynomial([0]) # y(x) = 0
        for i in range(len(self.x_vals)):
            ret = ret.add(self.basisPoly(i).scaled(self.y_vals[i]))
        self.poly = ret
    def basisPoly(self, j):
        p = self.numPoly(j)
        return p.scaled(1.0 / self.denomPoly(j))
    def denomPoly(self, j):
        prod = 1.0
        for i in range(len(self.x_vals)):
            if not i == j:
                prod *= self.x_vals[j] - self.x_vals[i]
        return prod
    def numPoly(self, j):
        prod = Polynomial([1]) # y(x) = 1
        for i in range(len(self.x_vals)):
            if not i == j:
                prod = prod.multiply(Polynomial([-self.x_vals[i], 1])) # y(x) = x - x_m
        return prod

def test_interp(points):
    # points = sample points to interpolate over
    x = arange(0.0, 10.0, 0.1)

def make_interp_element_graph(axis, plnt, elem, points, epoch_range):
    # epoch_range = valid epochs, [0, 1000] for example
    i = { 'a':0,'e':1,'i':2,'W':3,'w':4,'M':5 }[elem]
    axis.set_ylabel(elem)
    
    # Plot actual element history
    epochs = linspace(epoch_range[0], epoch_range[1], 1000)
    element_history = [plnt.osculating_elements(epoch(ei))[i] for ei in epochs]
    axis.plot(epochs, element_history, 'k--')
    
    # Plot attempted interpolation
    sampled_epochs = linspace(epoch_range[0], epoch_range[1], points)
    sampled_element_history = [plnt.osculating_elements( epoch(ei) )[i] for ei in sampled_epochs]
    axis.plot(sampled_epochs, sampled_element_history, 'ro') # plot sample points for reference
    e_interpltr = Interpolator(sampled_epochs, sampled_element_history)
    element_interp = [e_interpltr.getY(ei) for ei in epochs]
    axis.plot(epochs, element_interp, 'r:')

Conclusive thoughts¶

It is clear that interpolating over the elements for long periods does not work well with equidistant Lagrangian interpolation... Runge's phenomenon shows up, and there are some parts in W and w that are not smooth. The Wikipedia page on Lagrangian interpolation mentions using Chebyshev nodes to stop Runge's, but it wont fix the issue with the discontinuities in W and w.

It may be best to interpolate over the state (r, v) as Izzo suggested, since they should have cleaner curves over short periods, and clean oscillations over long periods. I don't know if the nuanced differences between the simple keplerian elements and the spice elements will show though. Would a piecewise Lagrangian work?

I will try interpolating over the state next.
